window.presCargarProductos = (catSel) => {
  const cat = catSel.value;
  const wrap = catSel.closest('.pres-desc-wrap');
  const prodSel = wrap.querySelector('.pres-prod-sel');
  if (!cat) {
    prodSel.innerHTML = '<option value="">-- Producto --</option>';
    prodSel.disabled = true;
    prodSel.style.background = 'var(--sand)';
    prodSel.style.color = 'var(--muted)';

  }

  prodSel.disabled = false;
  prodSel.style.background = '';
  prodSel.style.color = '';

};

**Function 7: presRecalcTotales** (TOT-01, TOT-02, TOT-03 — recalculates subtotal, discount, total)
```js
window.presRecalcTotales = () => {
  let subtotal = 0;
  document.querySelectorAll('.pres-linea').forEach(row => {
    const cant = parseFloat(row.querySelector('.pres-cant').value) || 0;
    const precio = parseFloat(row.querySelector('.pres-precio').value) || 0;
    subtotal += cant * precio;
  });
  const descValor = parseFloat(document.getElementById('pres-desc-valor').value) || 0;
  const esPct = document.getElementById('pres-desc-tipo-pct').classList.contains('active');
  let descuento = 0;
  if (esPct) {
    descuento = subtotal * (descValor / 100);
  } else {
    descuento = descValor;
  }
  descuento = Math.min(Math.max(descuento, 0), subtotal);
  const total = subtotal - descuento;
  document.getElementById('pres-val-subtotal').textContent = fmt(subtotal);
  document.getElementById('pres-val-descuento').textContent = '-' + fmt(descuento);
  document.getElementById('pres-val-total').textContent = fmt(total);
  document.getElementById('pres-row-descuento').style.display = descuento > 0 ? '' : 'none';
};
```
